Webpack module

  • Javascript module
  • Unique (module)
  • Can import / export
  • Local scope
// Imports
const FancyFramework = require('fancy-framework');

// Local scope
const localVar = 'app';

function mySuperApp() {
    return FancyFramework.makeAnApp(localVar).please();
}


// Exports
module.exports = mySuperApp;
